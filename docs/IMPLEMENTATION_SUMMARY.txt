INTEGRATION STEPS
═════════════════════════════════════════════════════════════════════

1. IMPORT COMPONENT
   import ShippingOptions from "@/components/checkout/ShippingOptions"

2. ADD TO CHECKOUT
   <ShippingOptions productId={id} areaCode={area} onSelect={fn} />

3. HANDLE SELECTION
   const [shipping, setShipping] = useState("normal")
   onSelect={(method) => setShipping(method)}

4. SUBMIT ORDER
   await createOrder({ shipping_method: shipping, ... })
